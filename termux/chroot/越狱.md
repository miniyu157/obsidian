进入 termux 环境

```bash
sudo chroot /proc/1/root /system/bin/sh -c ' \
    export PATH=/system/bin:/system/xbin; \
    exec /system/bin/run-as com.termux /data/data/com.termux/files/usr/bin/sh -c \
        ". /data/user/0/com.termux/files/usr/etc/termux/termux.env; exec /data/data/com.termux/files/usr/bin/login" \
'
```

进入 shell

```bash
sudo chroot /proc/1/root /system/bin/sh -c 'export PATH=/system/bin:/system/xbin; exec /system/bin/sh'
```

关闭屏幕

```bash
sudo chroot /proc/1/root /system/bin/sh -c "export PATH=/system/bin; /system/bin/input keyevent 26 > /dev/null"
```
